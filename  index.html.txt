    <script>
        function extractVideoSource(rawUrl) {
            // Check if string contains standard YouTube identifiers
            if (rawUrl.includes('://youtube.com') || rawUrl.includes('youtu.be/')) {
                let videoId = '';
                if(rawUrl.includes('://youtube.com')) {
                    videoId = rawUrl.split('v=')[1].split('&')[0];
                } else {
                    videoId = rawUrl.split('youtu.be/')[1].split('?')[0];
                }
                // Construct standard embed link configured with Autoplay & Mute extensions
                return { type: 'youtube', url: `https://youtube.com{videoId}?autoplay=1&mute=1&loop=1&playlist=${videoId}` };
            }
            // Default treatment assumes native web MP4 stream
            return { type: 'direct', url: rawUrl };
        }

        function launchArray() {
            const inputSrc = document.getElementById('videoUrl').value.trim();
            if(!inputSrc) {
                alert('Please parse a structural streaming link first.');
                return;
            }

            const mediaSource = extractVideoSource(inputSrc);
            
            // Populate all grid nodes with clean elements
            for(let i = 1; i <= 4; i++) {
                const targetBox = document.getElementById('box' + i);
                if(mediaSource.type === 'youtube') {
                    targetBox.innerHTML = `<iframe src="${mediaSource.url}" allow="autoplay; encrypted-media" allowfullscreen></iframe>`;
                } else {
                    targetBox.innerHTML = `<video src="${mediaSource.url}" controls autoplay loop muted></video>`;
                }
            }

            // Switch layout interfaces
            document.getElementById('setupPage').style.display = 'none';
            document.getElementById('dashboardHeader').style.display = 'flex';
            document.getElementById('gridPage').style.display = 'grid';
        }

        function resetArray() {
            // Flush running media content out of memory pools
            for(let i = 1; i <= 4; i++) {
                document.getElementById('box' + i).innerHTML = '';
            }
            // Toggle view visibility flags back
            document.getElementById('setupPage').style.display = 'block';
            document.getElementById('dashboardHeader').style.display = 'none';
            document.getElementById('gridPage').style.display = 'none';
        }
    </script>
